Step 3: If "tahun" mod 100 equals to 0 go to step 4, else go to step 5
Step 4: If "tahun" mod 400 equals to 0 go to step 5, else go to step 6
Step 5: Display "Tahun adalah kabisat."
Step 6: Display "Tahun bukan kabisat."

//Pseudocode
START
GET num tahun
IF {tahun % 4 = 0;
    IF {tahun % 100 = 0;
        IF {tahun % 400 = 0;
            PRINT "Tahun adalah kabisat."
        ELSE PRINT "Tahun bukan kabisat."
        }
    ELSE
        PRINT "Tahun adalah kabisat."
    }
ELSE
    PRINT "Tahun bukan kabisat."
}
END IF
END


3. Laundry Law

//Algorithm
Step 1: Initialize "laundry" to 20
Step 2: Initialize "counter" to 1
Step 3: While "counter" less than "laundry"
Step 4: Add 1 to "counter"
Step 5: Back to step 3
Step 6: Exit loop
Step 7: Display "counter"

//Pseudocode
START
SET laundry = 20
SET counter = 1
WHILE counter < laundry
    DO counter++
END WHILE
Print counter
END
